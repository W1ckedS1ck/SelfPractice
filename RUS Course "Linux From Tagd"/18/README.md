#v Задание 18.

#Написать скрипт генератор квази-слов с параметрами.

#Массив A содержит русские гласные буквы  (любые 5).
#Массив B содержит русские согласные буквы  (любые 15, кроме знаков).
#Слог состоит из случайных согласной и гласной. Именно в этом порядке.
#В слове $1 слогов (1<=$1<=5)
#Слоги накапливаются в переменной WORD.
#Слова выводятся на экран в stdout.
#Число слов $2 (3<=$2<=10).
#Если задан параметр $3, с вероятностью $3 (в процентах) слог состоит из гласной и согласной (меняется порядок)  (0<=$3<100).
#Подсказка  RANDOM % 100 < $3 
#Если скрипт запущен без параметров - выдать на stderr краткую справку (несколько строк) по использованию скрипта и параметров.
#Код завершения при этом должен быть 1 (help exit)
#Все переменные должны быть описаны.
#Если параметр вне диапазона - использовать ближайшее допустимое значение.

#!/bin/bash
if [[ $# -lt 2 || $# -gt 3 ]]; then
    echo "Использование: Введи кол-во слогов (1-5), введи кол-во слов (3-10) и по желанию процент инверсии (0-99)"
    exit 1
fi

GL=("а" "о" "у" "и" "э")
SO=("б" "в" "г" "д" "ж" "з" "к" "л" "м" "н" "п" "р" "с" "т" "ф")

S=$(($1 < 1 ? 1 : $1 > 5  ? 5   : $1))        # Если слогов меньше 1 - будет 1, если больше 5 - будет 5, иначе остается введенное число.
W=$(($2 < 3 ? 3 : $2 > 10 ? 10  : $2))        # Та же петрушка со словами. Диапазон 1-10.
if [[ -z "$3" ]]; then                        # Если не дан 3 аргумент - валится ошибка
    I=0
else
    I=$3
fi   
# I=${3:-0}                                   # Так тоже можно оказывается
I=$(($I < 0 ? 0 : $I > 100 ? 100 : $I))       # Проверяем, что инверсия в пределах 0-99

# Генерация слов
for (( i=0; i<W; i++ )); do
    WORD=""
    for (( j=0; j<S; j++ )); do
        GL=${GL[RANDOM % ${#GL[@]}]}
        SO=${SO[RANDOM % ${#SO[@]}]}
        ((RANDOM % 100 < I)) && WORD="$WORD$GL$SO" || WORD="$WORD$SO$GL"
    done
    echo $WORD
done
echo "ИТОГО $S слог(ов) $W слова(ов) с инверсией $I%"
